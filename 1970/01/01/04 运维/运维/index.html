安装

## 安装见官方文档     CENTOS安装: https://docs.docker.com/engine/install/centos/
yum-config-manager     --add-repo     https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
yum makecache fast
yum install docker-ce docker-ce-cli containerd.io
systemctl start docker
docker version

# 因为aws ec2 是amazon-linux-2 没有默认的可以安装 
# https://docs.docker.com/engine/install/binaries/ 
wget https://download.docker.com/linux/static/stable/x86_64/docker-20.10.6.tgz
tar xzvf docker-20.10.6.tgz
sudo cp docker/* /usr/bin/
sudo dockerd &

## ubuntu安装     官方文档 ： https://docs.docker.com/engine/install/ubuntu/ 
 sudo apt-get remove docker docker-engine docker.io containerd runc



================================================
#配置镜像加速器(内网才可以用 不配置也可以)    https://cr.console.aliyun.com/cn-hangzhou/instances/mirrors
#针对Docker客户端版本大于 1.10.0 的用户
#您可以通过修改daemon配置文件/etc/docker/daemon.json来使用加速器
sudo mkdir -p /etc/docker
sudo tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://bu0kmu2a.mirror.aliyuncs.com"]
}
EOF
sudo systemctl daemon-reload
sudo systemctl restart docker

常用命令

hub.docker.com /wudaxia/dd****0101

docker ps      # 活的容器
docker ps -a   # 所有的容器，包括历史的
docker images  #所有的镜像
docker rm XXX  #删除容器
docker rmi XXX #删除镜像
docker stop/start/restart/kill XXXX 
docker rm -f $(docker ps -aq)    # 删除所有的容器
docker ps -aq | xargs docker rm  # 删除所有的容器
docker run -it redis /bin/bash
docker run -d redis                 # 后台运行
docker run -d redis /bin/bash -c "while true;do echo hello;sleep 1;done"
# 常见的坑，docker容器后台启动后，必须要有一个前台进程，docker发现没有应用，就会自动停止
docker logs --help
docker logs -f -t -n 100 容器ID        ## 查看日志，  有的版本没有 -n参数
docker top 容器ID
docker inspect                # 查看容器ID
docker exec -it 容器ID /bin/bash           # 进入容器  查看配置 whereis nginx
docker attach 容器ID                       # 进入容器正在执行的终端，不会启动新的进程
docker cp 容器ID:/data/test.csv /mnt/d/code   # 从容器中copy文件出来
docker stats                     #查看容器状态
docker commit -m="提交的描述信息" -a="作者" 容器ID 目标镜像名：[TAG]
============================================================================
# 匿名挂载   -v 容器内路径
docker run -d -P --name nginx01 -v /etc/nginx nginx   # -P 随机分配端口 ， -v 不指定数据卷，随便
docker volume ls/rm/inspect/create/prune
# 具名挂载         -v 卷名：容器内路径
docker run -d -P --name nginx03 -v test:/etc/nginx nginx
# 设置权限         ro -> readonly      rw:readwrite    权限设置后只能通过宿主机改变
docker run -d -P --name nginx03 -v test:/etc/nginx:ro nginx
docker run -d -P --name nginx03 -v test:/etc/nginx:rw nginx

数据卷容器 (容器数据同步）

# 通过volumes-from实现容器间的数据共享
# 只要有一个容器还在，则数据还会存在
docker run -it --name nginx02 --volumes-from nginx01 daxia/nginx:1.0

# 建立两个数据库，共享数据卷
docker run -d -p 3306:3306 -v /home/daxia/mysql/conf:/etc/mysql/conf.d -v /home/daxia/mysql/data:/var/lib/mysql -e MYSQL_ROOT_PASSWORD=root123 --name=mysql01 mysql:5.7
docker run -d -p 3308:3306 -e MYSQL_ROOT_PASSWORD=root123 --volumes-from mysql01 --name mysql02 mysql:5.7
