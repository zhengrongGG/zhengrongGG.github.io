public Page<UdpGroupPart> listGroupPart() {
    Specification<UdpGroupPart> specification = new Specification<UdpGroupPart>() {
        @Override
        public Predicate toPredicate(Root<UdpGroupPart> root, CriteriaQuery<?> query, CriteriaBuilder cb) {
            List<Predicate> predicates = new LinkedList<>();
            Join<UdpGroupPart, UdpGroup> join = root.join("udpGroup", JoinType.INNER);
            predicatesList.add(cb.equal(root.get("isDelete"), UdpGroupPart.DELETE_NO));
            
            if (StringUtils.isNotBlank(name)) {
                Predicate nameLike = cb.like(root.get("name"), "%" + name + "%");
                if (StringUtils.isNumeric(name)) {
                    Predicate idEqual = cb.equal(root.get("id"), name);
                    Predicate one = cb.or(nameLike, idEqual);
                    predicatesList.add(one);
                } else {
                    predicatesList.add(nameLike);
                }
            }
            
            if (StringUtils.isNotBlank(status)) {
                CriteriaBuilder.In<Integer> one = cb.in(join.get("status"));
                Arrays.asList(status.split(",")).forEach(each -> {
                    one.value(Integer.parseInt(each));
                });
                predicatesList.add(one);
            }
            
            //排序操作 
            Order order = cb.asc(root.get("id"));
            query.orderBy(order);
            // query的构建可以认为是缺省的
            // query的构建可以认为是缺省的
            // query.where(predicatesList.toArray(new Predicate[predicatesList.size()]));
            return cb.and(predicatesList.toArray(new Predicate[predicatesList.size()]));
        }
    };
    return groupPartRepository.findAll(specification, PageRequest.of(page - 1, size, Sort.by(Sort.Direction.DESC, "id")));
}
